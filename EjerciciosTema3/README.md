Classe coche{

	privada entero puertas
	
	constructor(){
		puertas = 0
	}

	publico entero ObtenerPuertas(){
		retona puertas
	}
	
	publico vacio IncrementaPuertas(){
		puertas ++
	}

}

miCoche = nuevo coche

miCoche.IncrementarPuertas()

imprimir miCoche.ObtenerPuertas()
